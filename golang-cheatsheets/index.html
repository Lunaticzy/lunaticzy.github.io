// filename: main.go

}

流程控制

条件

if day == "sunday" || day == "saturday" {
  rest()
} else if day == "monday" && isTired() {
  groan()
} else {
  work()
}

If中的语句

if _, err := doThing(); err != nil {
  fmt.Println("Uh oh")
}

    if 语句中的条件可以放在 ; 之前的语句之前。,语句声明的变量只在if结束之前有效。

Switch语句

switch day {
  case "sunday":
    // cases don't "fall through" by default!
    fallthrough

  case "saturday":
    rest()

  default:
    work()
}

    Go里面switch默认相当于每个case最后带有break，匹配成功后不会自动向下执行其他case，而是跳出整个switch, 但是可以使用fallthrough强制执行后面的case代码

并发

协程 Goroutine

func main() {
  // 创建一个管道
  ch := make(chan string)

  // 启动协程
  go push("Moe", ch)
  go push("Larry", ch)
  go push("Curly", ch)

  // 接受结果
  // (因为协程是并发的,所以不能保证接收的顺序)
  fmt.Println(<-ch, <-ch, <-ch)
}

func push(name string, ch chan string) {
  msg := "Hey, " + name
  ch <- msg
}

    通道是并发安全的通信对象，在 goroutine 中使用

带缓存的管道 Buffered Channels

ch := make(chan int, 2)
ch <- 1
ch <- 2
ch <- 3
// fatal error:
// all goroutines are asleep - deadlock!
// 因为通道容量为2, 当超过2时死锁

关闭管道

ch <- 1
ch <- 2
ch <- 3
close(ch)  // 关闭一个管道

// 遍历一个管道, 直到关闭管道
for i := range ch {
  // do something... 
}


// 根据状态判断关闭
// closed if ok == false
v, ok := <- ch

WaitGroup

func main() {
  var wg sync.WaitGroup
  
  for _, item := range itemList {
    // Increment WaitGroup Counter
    wg.Add(1)
    go doOperation(item)
  }
  // Wait for goroutines to finish
  wg.Wait()
  
}

func doOperation(item string) {
  defer wg.Done()
  // do operation on item
  // ...
}

错误处理

Defer

func main() {
  defer fmt.Println("Done")
  fmt.Println("Working...")
}

    推迟运行一个函数，直到周围的函数返回。参数会立即计算，但函数调用直到稍后才会运行。

Deferring functions

func main() {
  defer func() {
    fmt.Println("Done")
  }()
  fmt.Println("Working...")
}

    Lambda更适合Defer哟

func main() {
  var d = int64(0)
  defer func(d *int64) {
    fmt.Printf("& %v Unix Sec\n", *d)
  }(&d)
  fmt.Print("Done ")
  d = time.Now().Unix()
}

    defer func 使用 d 的当前值，除非我们使用指针在 main 的末尾获取最终值。

Structs 结构体

定义

type Vertex struct {
  X int
  Y int
}

func main() {
  v := Vertex{1, 2}
  v.X = 4
  fmt.Println(v.X, v.Y) 
}

实例化

v := Vertex{X: 1, Y: 2}

// 字段名可以省略
v := Vertex{1, 2}

// Y是隐含的
v := Vertex{X: 1}

指向结构的指针

v := &Vertex{1, 2}
v.X = 2

    当 v 是一个指针时，做 v.X 和做 (*v).X 是一样的。

方法 Methods

接收器 Receivers

type Vertex struct {
   X, Y float64
 }

func (v Vertex) Abs() float64 {
  return math.Sqrt(v.X * v.X + v.Y * v.Y)
}

v := Vertex{1, 2}
v.Abs()

指针接收器 Mutation

func (v *Vertex) Scale(f float64) {
  v.X = v.X * f
  v.Y = v.Y * f
}

v := Vertex{6, 12}
v.Scale(0.5)
// `v` is updated

    定义一个指针接受器可以修改实例

接口 Interfaces

定义

type Shape interface {
  Area() float64
  Perimeter() float64
}

实现

// 定义一个结构体
type Rectangle struct {
  Length, Width float64
}

func (r Rectangle) Area() float64 {
  return r.Length * r.Width
}

func (r Rectangle) Perimeter() float64 {
  return 2 * (r.Length + r.Width)
}

    Struct Rectangle 通过实现Shape中所有方法来隐式实现接口 Shape。

实例

func main() {
  var r Shape = Rectangle{Length: 3, Width: 4}
   fmt.Printf("Type of r: %T, Area: %v, Perimeter: %v.", r, r.Area(), r.Perimeter())
}
